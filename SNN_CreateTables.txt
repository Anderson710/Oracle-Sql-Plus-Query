/*List of parent tables, that only has PK*/
CREATE TABLE Customer (
	Cust_ID 	VARCHAR(4) 	NOT NULL,
	Cust_Name 	VARCHAR(30) 	NOT NULL,
	Cust_Contact 	VARCHAR(11),
	Cust_Gender 	CHAR(1) 	NOT NULL,
	
	PRIMARY KEY(Cust_ID),

	CONSTRAINT chk_cust_id check (REGEXP_LIKE(Cust_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_name check (REGEXP_LIKE(Cust_Name, '^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_contact check (REGEXP_LIKE(Cust_Contact, '^\d{10,}$')),
	CONSTRAINT chk_gender check (UPPER(Cust_Gender) in ('M', 'F'))
);

CREATE TABLE Staff(
	Staff_ID 	VARCHAR(4) 	NOT NULL,
	Staff_Name 	VARCHAR(30) 	NOT NULL,
	Staff_Salary 	NUMBER(6,2) 	NOT NULL,
	Staff_Contact 	VARCHAR(11) 	NOT NULL,
	Date_Join 	DATE 		NOT NULL,
	Date_Birth 	DATE,
	Staff_Location 	VARCHAR(20),

	PRIMARY KEY(Staff_ID),

	CONSTRAINT chk_staff_id check (REGEXP_LIKE(Staff_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_staff_name check (REGEXP_LIKE(Staff_Name, '^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_staff_salary check (Staff_Salary >= 0),
	CONSTRAINT chk_staff_contact check (REGEXP_LIKE(Staff_Contact, '^(\d{10,}$)'))
);

CREATE TABLE Category(
	Cat_ID 		VARCHAR(4) 	NOT NULL,
	Cat_Name 	VARCHAR(20) 	NOT NULL,
	Cat_Desc 	VARCHAR(200), 
	Min_Price 	NUMBER(5,2) 	NOT NULL,
	Max_Price 	NUMBER(5,2) 	NOT NULL,

	PRIMARY KEY(Cat_ID),

	CONSTRAINT chk_cat_id check (REGEXP_LIKE(Cat_ID,'^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_cat_name check (REGEXP_LIKE(Cat_Name,'^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_min_price check (Min_Price >= 0),
	CONSTRAINT chk_max_price check (Max_Price >= 0)
);

CREATE TABLE Cancellation(
	Cancel_ID	VARCHAR(3)	NOT NULL,
	Cancel_NumItem	NUMBER(2)	NOT NULL,

	PRIMARY KEY(Cancel_ID),

	CONSTRAINT chk_cancel_id check (REGEXP_LIKE(Cancel_ID, '^([A-Z]{1}+)([0-9]{2})$')),
	CONSTRAINT chk_numitem check (Cancel_NumItem >= 0)
);

CREATE TABLE Refund(
	Ref_ID 		VARCHAR(3) 	NOT NULL,
	Ref_NumItem 	NUMBER(2)	NOT NULL,
	Ref_Remarks	VARCHAR(50)	NOT NULL,

	PRIMARY KEY(Ref_ID),

	CONSTRAINT chk_ref_id check (REGEXP_LIKE(Ref_ID, '^([A-Z]{1}+)([0-9]{2})$')),
	CONSTRAINT chk_ref_numitem check (Ref_NumItem>= 0)
);

CREATE TABLE Promotion(
	Promo_ID	VARCHAR(4) 	NOT NULL,
	Promo_Name	VARCHAR(20)	NOT NULL,
	Promo_Type	VARCHAR(10) 	NOT NULL,
	Promo_Desc	VARCHAR(50)	NOT NULL,

	PRIMARY KEY(Promo_ID),

	CONSTRAINT chk_id check (REGEXP_LIKE(Promo_ID, '^([A-Z]{1}+)([0-9]{3})$'))
);

CREATE TABLE Takeaway(
	Take_ID		VARCHAR(4)	NOT NULL,
	Take_Charges	NUMBER(4,2)	NOT NULL,
	Take_Time	VARCHAR(10) 	NOT NULL,
	Take_Type	VARCHAR(20) 	NOT NULL,

	PRIMARY KEY(Take_ID),

	CONSTRAINT chk_take_id check (REGEXP_LIKE(Take_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_take_charges check (Take_Charges >= 0)
);

CREATE TABLE Supplier(
	Supp_ID  	VARCHAR(4) 	NOT NULL,
	Supp_Name 	VARCHAR(30) 	NOT NULL,
	Supp_Type 	VARCHAR(10) 	NOT NULL,
	Supp_Contact 	VARCHAR(11) 	NOT NULL,
	Supp_Email 	VARCHAR(40),
	Supp_Address 	VARCHAR(50) 	NOT NULL,

	PRIMARY KEY(Supp_ID),

	CONSTRAINT chk_supp_id check (REGEXP_LIKE(Supp_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_supp_name check (REGEXP_LIKE(Supp_Name, '^(([a-zA-Z]\s*\.*){2,})$')),
	CONSTRAINT chk_supp_contact check (REGEXP_LIKE(Supp_Contact, '^\d{10,}$')),
	CONSTRAINT chk_supp_email check (REGEXP_LIKE(Supp_Email,'^[a-zA-Z]\w+@(\S+)$'))
);

CREATE TABLE Raw_Ingredient(
	Ing_ID 		VARCHAR(4) 	NOT NULL,
	Ing_Name 	VARCHAR(15) 	NOT NULL,
	Ing_Qty	 	NUMBER(3) 	NOT NULL,
	Ing_Unit 	VARCHAR(10) 	NOT NULL,
	Ing_QtyReq 	NUMBER(3) 	NOT NULL,

	PRIMARY KEY(Ing_ID),

	CONSTRAINT chk_ing_id check (REGEXP_LIKE(Ing_ID,'^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_ing_name check (REGEXP_LIKE(Ing_Name,'^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_ing_qty check (Ing_Qty >= 0),
	CONSTRAINT chk_ing_unit check (REGEXP_LIKE(Ing_Unit,'^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_ing_qtyReq check (Ing_QtyReq >= 0)
);

/*List of Child Tables, with PK and FK*/
CREATE TABLE Menu(
	Item_ID 	VARCHAR(4) 	NOT NULL,
	Item_Name 	VARCHAR(30) 	NOT NULL,
	Item_Desc 	VARCHAR(200),
	Item_Price 	NUMBER(5,2) 	NOT NULL,
	Date_Intro 	DATE 		NOT NULL,
	Cat_ID 		VARCHAR(4) 	NOT NULL,
	Menu_Remarks 	VARCHAR(100),

	PRIMARY KEY(Item_ID),
	FOREIGN KEY(Cat_ID) REFERENCES Category(Cat_ID),

	CONSTRAINT chk_item_id check (REGEXP_LIKE(Item_ID,'^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_item_name check (REGEXP_LIKE(Item_Name,'^(([a-zA-Z]\s*){2,})$'))
);

CREATE TABLE Menu_Ingredient_List(
	Item_ID 	VARCHAR(4) 	NOT NULL,
	Ing_ID 		VARCHAR(4) 	NOT NULL,
	Usage_Qty 	NUMBER(5,2) 	NOT NULL,
	Usage_Unit 	VARCHAR(10) 	NOT NULL,

	PRIMARY KEY(Item_ID,Ing_ID),
	FOREIGN KEY(Item_ID) REFERENCES Menu(Item_ID),
	FOREIGN KEY(Ing_ID) REFERENCES Raw_Ingredient(Ing_ID),

	CONSTRAINT chk_usage_qty check (Usage_Qty >= 0),
	CONSTRAINT chk_usage_unit check (REGEXP_LIKE(Usage_Unit,'^(([a-zA-Z]\s*){2,})$'))
);

CREATE TABLE Supplier_Ingredient_List(
	Supp_ID 	VARCHAR(4) 	NOT NULL,
	Ing_ID 		VARCHAR(4) 	NOT NULL,
	Supp_Qty 	NUMBER(4) 	NOT NULL,
	Supp_Unit 	VARCHAR(10) 	NOT NULL,
	Supp_UPrice 	NUMBER(5,2) 	NOT NULL,

	PRIMARY KEY(Supp_ID, Ing_ID),
	FOREIGN KEY(Supp_ID) REFERENCES Supplier(Supp_ID),
	FOREIGN KEY(Ing_ID) REFERENCES Raw_Ingredient(Ing_ID),

	CONSTRAINT chk_supp_qty check (Supp_Qty >= 0),
	CONSTRAINT chk_supp_unit check (REGEXP_LIKE(Supp_Unit,'^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_supp_uprice check (Supp_UPrice >= 0)
);

CREATE TABLE Requisition_Purchase(
	Purc_ID 	VARCHAR(4) 	NOT NULL,
	Staff_ID 	VARCHAR(4) 	NOT NULL,
	Purc_Date 	DATE 		NOT NULL,
	Supp_ID 	VARCHAR(4) 	NOT NULL,
	Purc_Status 	VARCHAR(10) 	NOT NULL,
	Num_Requisition NUMBER(3) 	NOT NULL,
	Purc_Amt 	NUMBER(6,2) 	NOT NULL,
	
	PRIMARY KEY(Purc_ID),
	FOREIGN KEY(Staff_ID) REFERENCES Staff(Staff_ID),
	FOREIGN KEY(Supp_ID) REFERENCES	Supplier(Supp_ID),

	CONSTRAINT chk_purc_id check (REGEXP_LIKE(Purc_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_num_requisition check (Num_Requisition >= 0),
	CONSTRAINT chk_purc_amt check (Purc_Amt >= 0)
);

CREATE TABLE Requisition_List(
	Purc_ID 	VARCHAR(4) 	NOT NULL,
	Ing_ID 		VARCHAR(4) 	NOT NULL,
	ReqItem_Qty 	NUMBER(4) 	NOT NULL,
	ReqItem_Unit 	VARCHAR(10) 	NOT NULL,
	ReqItem_UPrice 	NUMBER(5,2) 	NOT NULL,
	ReqItem_Subttl 	NUMBER(10,2) 	NOT NULL,

	PRIMARY KEY(Purc_ID, Ing_ID),
	FOREIGN KEY(Purc_ID) REFERENCES Requisition_Purchase(Purc_ID),
	FOREIGN KEY(Ing_ID) REFERENCES Raw_Ingredient(Ing_ID),	

	CONSTRAINT chk_reqItem_qty check (ReqItem_Qty >= 0),
	CONSTRAINT chk_reqItem_Unit check (REGEXP_LIKE(ReqItem_Unit,'^(([a-zA-Z]\s*){2,})$')),
	CONSTRAINT chk_reqItem_Uprice check (ReqItem_UPrice >= 0),
	CONSTRAINT chk_reqItem_Subttl check (ReqItem_Subttl >= 0)
);

CREATE TABLE Menu_Order(
	Ord_ID 		VARCHAR(4) 	NOT NULL,
	Cust_ID 	VARCHAR(4)  	NOT NULL,
	Ord_Date 	DATE 		NOT NULL,
	Ord_Time 	VARCHAR(10)  	NOT NULL,
	Staff_ID 	VARCHAR(4) 	NOT NULL,
	Ord_Type 	VARCHAR(8) 	NOT NULL,
	Num_Item 	NUMBER(4) 	NOT NULL,	
	Promo_ID 	VARCHAR(4),
	Ord_Amt  	NUMBER(6,2) 	NOT NULL,
	Take_ID 	VARCHAR(4),

	PRIMARY KEY(Ord_ID),
	FOREIGN KEY(Cust_ID) REFERENCES Customer(Cust_ID),
	FOREIGN KEY(Staff_ID) REFERENCES Staff(Staff_ID),
	FOREIGN KEY(Promo_ID) REFERENCES Promotion(Promo_ID),
	FOREIGN KEY(Take_ID) REFERENCES Takeaway(Take_ID),

	CONSTRAINT chk_ord_id check (REGEXP_LIKE(Ord_ID, '^([A-Z]{1}+)([0-9]{3})$')),
	CONSTRAINT chk_num_item check (Num_Item >= 0),
	CONSTRAINT chk_ord_amount check (Ord_Amt >= 0)
);

CREATE TABLE Order_Item (
	Ord_ID 		VARCHAR(4) 	NOT NULL,
	Item_ID 	VARCHAR(4) 	NOT NULL,
	OrdItem_Qty 	NUMBER(3) 	NOT NULL,
	Promo_ID 	VARCHAR(4),
	OrdItem_Subttl 	NUMBER(6,2),	
	Cancel_ID 	VARCHAR(3),
	Ref_ID 	 	VARCHAR(3),

	PRIMARY KEY(Ord_ID,Item_ID),
	FOREIGN KEY(Ord_ID) REFERENCES Menu_Order (Ord_ID),
	FOREIGN KEY(Item_ID) REFERENCES Menu (Item_ID),
	FOREIGN KEY(Promo_ID) REFERENCES Promotion (Promo_ID),
	FOREIGN KEY(Cancel_ID) REFERENCES Cancellation (Cancel_ID),
	FOREIGN KEY(Ref_ID) REFERENCES Refund (Ref_ID),

	CONSTRAINT chk_orditem_qty check (OrdItem_Qty >= 0),
	CONSTRAINT chk_orditem_subttl check (OrdItem_Subttl >= 0)
);
